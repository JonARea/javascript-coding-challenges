function largestOfFour(arr) {
    var largest4 = [];
    function Amax(array) {
     return Math.max.apply(Math, array);
    }
    for (var i = 0; i < arr.length; i++) {
       
      largest4.push(Amax(arr[i]));
    }
    console.log(Amax(arr[0]));
    return largest4;
}
