function destroyer(arr) {
  var args = Array.prototype.slice.call(arguments);
  function func(value) {
    for (var j = 1; j <= args.length; j++) {
        if (value == args[j]) {
          return false;
        }
    }
    return true;
  }
  return arr.filter(func);
    
}
destroyer([1, 2, 3, 1, 2, 3], 2, 3);
